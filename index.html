} // ═══════════════════════════════════════════════════════════ // HELPER FUNCTIONS // ═══════════════════════════════════════════════════════════ /** * Get the header row (row 1) from a sheet as a flat array of strings */ function getHeaderRow(sheet) { const lastCol = sheet.getLastColumn(); if (lastCol === 0) return []; return sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim()); } /** * Build a lowercase column map: { 'header name': columnIndex } */ function buildColMap(headers) { const map = {}; headers.forEach((h, idx) => { if (h) map[h.toLowerCase()] = idx; }); return map; } /** * Find a column index by trying multiple possible header names * (case-insensitive exact match, then partial/includes match) */ function findCol(colMap, ...names) { // Exact match first for (const name of names) { const key = name.toLowerCase(); if (colMap[key] !== undefined) return colMap[key]; } // Partial match const keys = Object.keys(colMap); for (const name of names) { const lower = name.toLowerCase(); const found = keys.find(k => k.includes(lower) || lower.includes(k)); if (found !== undefined) return colMap[found]; } return -1; } /** * Set a value in a row array by looking up the column from the map. * Handles multiple possible header names. */ function setCol(rowArray, colMap, headerName, value) { if (value === undefined || value === null) return; // Try exact match const idx = findCol(colMap, headerName); if (idx >= 0 && idx < rowArray.length) { rowArray[idx] = value; } } /** * Write a value to a specific sheet cell, looking up column by header name. */ function setSheetCell(sheet, rowNum, colMap, headerName, value) { if (value === undefined || value === null) return; // Allow empty strings (e.g. clearing billing fields) const idx = findCol(colMap, headerName); if (idx >= 0) { sheet.getRange(rowNum, idx + 1).setValue(value); } } /** * Copy formulas from a reference cycle row into a newly inserted row. * Finds an existing cycle row in the same pod sheet to use as a template. * Only copies cells that contain formulas (skips plain values). * Adjusts row references in formulas to point to the new row. * * @param {Sheet} sheet - The sheet containing both rows * @param {number} newRowNum - The 1-indexed row number of the newly inserted row * @param {number} refRowNum - The 1-indexed row number of the reference row to copy formulas from * @param {number} numCols - Total number of columns */ function copyFormulasFromRow(sheet, newRowNum, refRowNum, numCols) { if (refRowNum < 1 || newRowNum < 1) return; const refRange = sheet.getRange(refRowNum, 1, 1, numCols); const formulas = refRange.getFormulas()[0]; // array of formula strings (empty string if no formula) for (var col = 0; col < numCols; col++) { if (formulas[col] && formulas[col] !== '') { // Adjust the formula: replace references to the old row number with the new row number var adjusted = formulas[col].replace(new RegExp('\\b' + refRowNum + '\\b', 'g'), String(newRowNum)); try { sheet.getRange(newRowNum, col + 1).setFormula(adjusted); } catch (e) { // If formula adjustment fails, try copying as-is try { sheet.getRange(newRowNum, col + 1).setFormula(formulas[col]); } catch (e2) { /* skip */ } } } } } /** * Find a reference cycle row in the sheet to copy formulas from. * Looks for any row where col B has "Cycle" in it (a cycle row). * Prefers rows near the target area, but will take any cycle row. * * @param {Sheet} sheet - The sheet to search * @param {number} nearRow - 0-indexed row to search near (e.g. manager header) * @returns {number} 1-indexed row number of reference cycle row, or -1 if not found */ function findReferenceCycleRow(sheet, nearRow) { const allData = sheet.getDataRange().getValues(); // First pass: look within 50 rows of the target for (var i = Math.max(1, nearRow - 25); i < Math.min(allData.length, nearRow + 50); i++) { var colB = String(allData[i][1]).trim().toLowerCase(); if (colB.indexOf('cycle') === 0 && colB !== '') { return i + 1; // convert to 1-indexed } } // Second pass: anywhere in the sheet for (var i = 1; i < allData.length; i++) { var colB = String(allData[i][1]).trim().toLowerCase(); if (colB.indexOf('cycle') === 0 && colB !== '') { return i + 1; } } return -1; // no reference found }
