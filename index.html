/** * ═══════════════════════════════════════════════════════════ * RoofIgnite Dashboard — Write-Back Apps Script * ═══════════════════════════════════════════════════════════ * * This script handles ALL write operations from the * RoofIgnite Command Center dashboard back to Google Sheets. * * SETUP: * 1. Open your Google Sheet * 2. Extensions → Apps Script * 3. Delete any existing code in Code.gs * 4. Paste this ENTIRE script * 5. Click Save (Ctrl+S) * 6. Deploy → New deployment * - Type: Web app * - Execute as: Me * - Who has access: Anyone * 7. Click Deploy, authorize when prompted * 8. Copy the Web app URL → paste into dashboard settings * * ACTIONS HANDLED (8 total): * ─────────────────────────────────────────────────────── * setManagerGregMode → Greg Config tab (manager row) * setAccountGregMode → Greg Config tab (account row) * createClient → Inserts new account row below manager header * updateManager → Moves account to new manager * toggleStatus → Pauses/activates an account * updateCycle → Updates goals/budgets on a cycle row * addCycle → Appends a new cycle row for an account * addManager → Creates new manager header row (blue) + Greg Config * ─────────────────────────────────────────────────────── * * IMPORTANT: After making changes to this script, you must * create a NEW deployment version for changes to take effect. * (Deploy → Manage deployments → Edit → New version → Deploy) */ // ═══════════════════════════════════════════════════════════ // CONFIGURATION // ═══════════════════════════════════════════════════════════ // Sheet tab names — adjust if yours are different const POD_SHEETS = [ 'Pod 2 - RoofIgnite', 'Pod 1 - RoofIgnite', 'ContractorsIgnite' ]; // Sub-section labels that are NOT manager names // Used to distinguish manager header rows from sub-section headers const KNOWN_SUB_SECTIONS = [ 'roofignite', 'roof ignite', 'roofers ignite', 'hvac ignite', 'contractorsignite', 'contractors ignite', 'cign ignite', 'solar ignite', 'pending', 'expansion', 'active', 'inactive', 'paused', 'pause', 'winter', 'kpi' ]; // Manager header background color (blue) const MANAGER_HEADER_COLOR = '#4285f4'; // ═══════════════════════════════════════════════════════════ // MAIN ENTRY POINT // ═══════════════════════════════════════════════════════════ function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const data = JSON.parse(e.postData.contents); const action = data.action; switch (action) { case 'setManagerGregMode': case 'setAccountGregMode': handleGregConfig(ss, action, data); break; case 'createClient': handleCreateClient(ss, data); break; case 'updateManager': handleUpdateManager(ss, data); break; case 'toggleStatus': handleToggleStatus(ss, data); break; case 'updateCycle': handleUpdateCycle(ss, data); break; case 'addCycle': handleAddCycle(ss, data); break; case 'addManager': handleAddManager(ss, data); break; default: return jsonResponse({ ok: false, error: 'Unknown action: ' + action }); } return jsonResponse({ ok: true, action: action }); } catch (err) { return jsonResponse({ ok: false, error: err.message }); } } // Allow GET requests for testing (optional) function doGet(e) { return jsonResponse({ status: 'RoofIgnite Dashboard API is running', timestamp: new Date().toISOString() }); } function jsonResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ═══════════════════════════════════════════════════════════ // 1. GREG CONFIG (setManagerGregMode / setAccountGregMode) // ═══════════════════════════════════════════════════════════ function handleGregConfig(ss, action, data) { // Get or create the Greg Config sheet let configSheet = ss.getSheetByName('Greg Config'); if (!configSheet) { configSheet = ss.insertSheet('Greg Config'); configSheet.appendRow(['Type', 'Name', 'Mode']); configSheet.getRange('A1:C1').setFontWeight('bold'); } const type = (action === 'setManagerGregMode') ? 'manager' : 'account'; const name = data.manager || data.name; const mode = data.mode; // 'HARD', 'SOFT', 'OFF', or null (to remove) // Search for existing row const vals = configSheet.getDataRange().getValues(); let found = false; for (let i = 1; i < vals.length; i++) { if (String(vals[i][0]).trim() === type && String(vals[i][1]).trim() === name) { if (mode === null || mode === 'null') { // Remove the override row entirely configSheet.deleteRow(i + 1); } else { // Update existing row configSheet.getRange(i + 1, 3).setValue(mode); } found = true; break; } } // If not found and we have a mode to set, append new row if (!found && mode !== null && mode !== 'null') { configSheet.appendRow([type, name, mode]); } } // ═══════════════════════════════════════════════════════════ // 2. CREATE CLIENT (createClient) // ═══════════════════════════════════════════════════════════ function handleCreateClient(ss, data) { const podName = data.pod || 'Pod 2 - RoofIgnite'; const sheet = ss.getSheetByName(podName); if (!sheet) throw new Error('Sheet not found: ' + podName); // Find the header row and build column map const headers = getHeaderRow(sheet); const colMap = buildColMap(headers); const numCols = sheet.getLastColumn(); const newRow = new Array(numCols).fill(''); // Column A (index 0) = Account name newRow[0] = data.name || ''; // Column B (index 1) = Cycle label (empty for new account header) newRow[1] = ''; // Column C (index 2) = Ad Account ID newRow[2] = data.adAccountId || ''; // Fill in mapped columns setCol(newRow, colMap, 'account manager', data.manager); setCol(newRow, colMap, 'booked appointment goal', data.bookedGoal); setCol(newRow, colMap, 'greg goal', data.gregGoal); setCol(newRow, colMap, 'cpa goal', data.cpaGoal); setCol(newRow, colMap, 'daily budget', data.dailyBudget); setCol(newRow, colMap, 'monthly budget', data.monthlyBudget); // ── Find the manager's header row and insert right below it ── const managerName = data.manager; const allData = sheet.getDataRange().getValues(); let managerHeaderRow = -1; if (managerName) { for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); const colB = String(allData[i][1]).trim(); // Manager header: col A has text, col B is empty, not a known sub-section if (colA && !colB) { const lower = colA.toLowerCase(); if (KNOWN_SUB_SECTIONS.includes(lower) || lower.startsWith('[paused]')) continue; if (lower === managerName.toLowerCase()) { managerHeaderRow = i; // 0-indexed break; } } } } if (managerHeaderRow >= 0) { // Insert right after the manager's header row (top-down, newest first) const insertRow = managerHeaderRow + 2; // +1 for 0→1 index, +1 for "after" sheet.insertRowBefore(insertRow); sheet.getRange(insertRow, 1, 1, numCols).setValues([newRow]); } else { // Fallback: couldn't find the manager header, append to end sheet.appendRow(newRow); } } // ═══════════════════════════════════════════════════════════ // 3. UPDATE MANAGER (updateManager) // ═══════════════════════════════════════════════════════════ function handleUpdateManager(ss, data) { const accountName = data.name; const newManager = data.manager; // Search all pod sheets for rows belonging to this account for (const podName of POD_SHEETS) { const sheet = ss.getSheetByName(podName); if (!sheet) continue; const headers = getHeaderRow(sheet); const colMap = buildColMap(headers); const mgrColIdx = findCol(colMap, 'account manager', 'manager', 'acct manager'); if (mgrColIdx < 0) continue; const allData = sheet.getDataRange().getValues(); let inAccount = false; for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); // Check if this is the account's header row if (colA === accountName) { inAccount = true; // Update manager on header row sheet.getRange(i + 1, mgrColIdx + 1).setValue(newManager); continue; } // If we're inside this account's rows (cycle rows have empty col A, filled col B) if (inAccount) { if (colA !== '' && colA !== accountName) { // Hit a different account name → stop break; } // Update manager on cycle rows too sheet.getRange(i + 1, mgrColIdx + 1).setValue(newManager); } } if (inAccount) return; // Found and updated } } // ═══════════════════════════════════════════════════════════ // 4. TOGGLE STATUS (toggleStatus) // ═══════════════════════════════════════════════════════════ function handleToggleStatus(ss, data) { const accountName = data.name; const isPaused = data.isPaused; // The dashboard tracks pause state. In the sheet, we'll mark it // by setting the "Status" or "Notes" column. We'll look for a // status-like column first, otherwise use notes. for (const podName of POD_SHEETS) { const sheet = ss.getSheetByName(podName); if (!sheet) continue; const headers = getHeaderRow(sheet); const colMap = buildColMap(headers); // Try to find a status column let statusCol = findCol(colMap, 'status', 'account status'); // Fallback: use notes column if (statusCol < 0) statusCol = findCol(colMap, 'notes', 'note'); const allData = sheet.getDataRange().getValues(); for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); if (colA === accountName) { if (statusCol >= 0) { const currentVal = String(allData[i][statusCol]).trim(); if (isPaused) { // Mark as paused — prepend [PAUSED] if not already there if (!currentVal.includes('[PAUSED]')) { sheet.getRange(i + 1, statusCol + 1).setValue('[PAUSED] ' + currentVal); } } else { // Remove [PAUSED] marker sheet.getRange(i + 1, statusCol + 1).setValue(currentVal.replace(/\[PAUSED\]\s*/g, '')); } } return; } } } } // ═══════════════════════════════════════════════════════════ // 5. UPDATE CYCLE (updateCycle) // ═══════════════════════════════════════════════════════════ function handleUpdateCycle(ss, data) { const accountName = data.name; const cycleName = data.cycle; for (const podName of POD_SHEETS) { const sheet = ss.getSheetByName(podName); if (!sheet) continue; const headers = getHeaderRow(sheet); const colMap = buildColMap(headers); const allData = sheet.getDataRange().getValues(); let inAccount = false; for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); const colB = String(allData[i][1]).trim(); // Track which account we're in if (colA !== '') { inAccount = (colA === accountName); } // Look for the matching cycle row if (inAccount && colB === cycleName) { // Update each field that was sent setSheetCell(sheet, i + 1, colMap, 'cycle start date', data.cycleStartDate); setSheetCell(sheet, i + 1, colMap, 'cycle end date', data.cycleEndDate); setSheetCell(sheet, i + 1, colMap, 'booked appointment goal', data.bookedGoal); setSheetCell(sheet, i + 1, colMap, 'greg goal', data.gregGoal); setSheetCell(sheet, i + 1, colMap, 'cpa goal', data.cpaGoal); setSheetCell(sheet, i + 1, colMap, 'daily budget', data.dailyBudget); setSheetCell(sheet, i + 1, colMap, 'monthly budget', data.monthlyBudget); return; } } } } // ═══════════════════════════════════════════════════════════ // 6. ADD CYCLE (addCycle) // ═══════════════════════════════════════════════════════════ function handleAddCycle(ss, data) { const accountName = data.account; const podName = data.pod || 'Pod 2 - RoofIgnite'; // Try the specified pod first, then search all pods const sheetsToSearch = [podName, ...POD_SHEETS.filter(p => p !== podName)]; for (const sheetName of sheetsToSearch) { const sheet = ss.getSheetByName(sheetName); if (!sheet) continue; const headers = getHeaderRow(sheet); const colMap = buildColMap(headers); const allData = sheet.getDataRange().getValues(); // Find the account and its last cycle row let accountStartRow = -1; let lastAccountRow = -1; for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); if (colA === accountName) { accountStartRow = i; lastAccountRow = i; continue; } if (accountStartRow >= 0) { // Still in this account's cycle rows (col A empty, col B has cycle name) if (colA === '' && String(allData[i][1]).trim() !== '') { lastAccountRow = i; } else if (colA !== '') { // Hit the next account break; } } } if (accountStartRow < 0) continue; // Account not found in this sheet // Build the new cycle row const numCols = sheet.getLastColumn(); const newRow = new Array(numCols).fill(''); // Col A empty (account name only on first row), Col B = cycle name newRow[0] = ''; newRow[1] = data.cycle || ''; newRow[2] = data.adAccountId || ''; // Map fields to columns setCol(newRow, colMap, 'cycle start date', data.cycleStartDate); setCol(newRow, colMap, 'cycle end date', data.cycleEndDate); setCol(newRow, colMap, 'booked appointment goal', data.bookedGoal); setCol(newRow, colMap, 'greg goal', data.gregGoal); setCol(newRow, colMap, 'cpa goal', data.cpaGoal); setCol(newRow, colMap, 'daily budget', data.dailyBudget); setCol(newRow, colMap, 'monthly budget', data.monthlyBudget); setCol(newRow, colMap, 'account manager', data.manager); // Insert the row right after the last row belonging to this account const insertAfterRow = lastAccountRow + 1; // 0-indexed → sheet row sheet.insertRowAfter(insertAfterRow + 1); // insertRowAfter is 1-indexed const targetRow = insertAfterRow + 2; // the newly inserted row (1-indexed) sheet.getRange(targetRow, 1, 1, numCols).setValues([newRow]); return; } // Fallback: if account not found anywhere, append to the specified pod sheet const fallbackSheet = ss.getSheetByName(podName); if (fallbackSheet) { const headers = getHeaderRow(fallbackSheet); const colMap = buildColMap(headers); const numCols = fallbackSheet.getLastColumn(); const newRow = new Array(numCols).fill(''); newRow[0] = accountName; newRow[1] = data.cycle || ''; newRow[2] = data.adAccountId || ''; setCol(newRow, colMap, 'cycle start date', data.cycleStartDate); setCol(newRow, colMap, 'cycle end date', data.cycleEndDate); setCol(newRow, colMap, 'booked appointment goal', data.bookedGoal); setCol(newRow, colMap, 'greg goal', data.gregGoal); setCol(newRow, colMap, 'cpa goal', data.cpaGoal); setCol(newRow, colMap, 'daily budget', data.dailyBudget); setCol(newRow, colMap, 'monthly budget', data.monthlyBudget); setCol(newRow, colMap, 'account manager', data.manager); fallbackSheet.appendRow(newRow); } } // ═══════════════════════════════════════════════════════════ // 7. ADD MANAGER (addManager) // ═══════════════════════════════════════════════════════════ function handleAddManager(ss, data) { const managerName = data.name; const podName = data.pod || 'Pod 2 - RoofIgnite'; const sheet = ss.getSheetByName(podName); if (!sheet) throw new Error('Sheet not found: ' + podName); // Find the right insertion point: // After the last existing manager's section, before Paused/footer sections const allData = sheet.getDataRange().getValues(); const numCols = sheet.getLastColumn(); let lastManagerRow = -1; // last row belonging to the last real manager let pausedSectionRow = -1; // row where "Paused" or similar footer starts for (let i = 1; i < allData.length; i++) { const colA = String(allData[i][0]).trim(); const colB = String(allData[i][1]).trim(); if (!colA && !colB) continue; // empty row // Section marker: col A has text, col B is empty if (colA && !colB) { const lower = colA.toLowerCase(); if (KNOWN_SUB_SECTIONS.includes(lower) || lower.startsWith('[paused]')) { // Check if it's specifically a "paused" or end-section marker if (lower === 'paused' || lower === 'pause' || lower === 'winter' || lower === 'inactive') { if (pausedSectionRow < 0) pausedSectionRow = i; } continue; // skip sub-sections } // It's a manager name — track it lastManagerRow = i; continue; } // Data row (account or cycle) — extends the current manager's section if (lastManagerRow >= 0) { lastManagerRow = i; } } // Determine where to insert let insertRow; if (pausedSectionRow >= 0) { // Insert right before the Paused section insertRow = pausedSectionRow + 1; // +1 for 1-indexed } else if (lastManagerRow >= 0) { // Insert after the last manager's last row insertRow = lastManagerRow + 2; // +1 for 0→1 index, +1 for "after" } else { // Empty sheet — just append after header insertRow = sheet.getLastRow() + 1; } // Insert a blank row and write the manager name sheet.insertRowBefore(insertRow); const newRow = new Array(numCols).fill(''); newRow[0] = managerName; sheet.getRange(insertRow, 1, 1, numCols).setValues([newRow]); // Style the manager header row: blue background, white bold text const headerRange = sheet.getRange(insertRow, 1, 1, numCols); headerRange.setBackground(MANAGER_HEADER_COLOR); headerRange.setFontColor('#ffffff'); headerRange.setFontWeight('bold'); // Also add to Greg Config as SOFT by default handleGregConfig(ss, 'setManagerGregMode', { manager: managerName, mode: 'SOFT' }); } // ═══════════════════════════════════════════════════════════ // HELPER FUNCTIONS // ═══════════════════════════════════════════════════════════ /** * Get the header row (row 1) from a sheet as a flat array of strings */ function getHeaderRow(sheet) { const lastCol = sheet.getLastColumn(); if (lastCol === 0) return []; return sheet.getRange(1, 1, 1, lastCol).getValues()[0].map(h => String(h).trim()); } /** * Build a lowercase column map: { 'header name': columnIndex } */ function buildColMap(headers) { const map = {}; headers.forEach((h, idx) => { if (h) map[h.toLowerCase()] = idx; }); return map; } /** * Find a column index by trying multiple possible header names * (case-insensitive exact match, then partial/includes match) */ function findCol(colMap, ...names) { // Exact match first for (const name of names) { const key = name.toLowerCase(); if (colMap[key] !== undefined) return colMap[key]; } // Partial match const keys = Object.keys(colMap); for (const name of names) { const lower = name.toLowerCase(); const found = keys.find(k => k.includes(lower) || lower.includes(k)); if (found !== undefined) return colMap[found]; } return -1; } /** * Set a value in a row array by looking up the column from the map. * Handles multiple possible header names. */ function setCol(rowArray, colMap, headerName, value) { if (value === undefined || value === null) return; // Try exact match const idx = findCol(colMap, headerName); if (idx >= 0 && idx < rowArray.length) { rowArray[idx] = value; } } /** * Write a value to a specific sheet cell, looking up column by header name. */ function setSheetCell(sheet, rowNum, colMap, headerName, value) { if (value === undefined || value === null) return; const idx = findCol(colMap, headerName); if (idx >= 0) { sheet.getRange(rowNum, idx + 1).setValue(value); } }
